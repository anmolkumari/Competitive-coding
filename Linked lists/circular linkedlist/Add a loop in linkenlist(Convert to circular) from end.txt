
package link;


public class Link {
    Node Head=null;

       Node current;

    class Node{
       Node next;
       int data;
       Node(int d){
           data=d;
           next=null;
       }}

      void insert(int new_node){

       if(Head==null){

           Node node=new Node(new_node);
           Head=node;
          current=Head;

       }
       else{

           Node new_n=new Node(new_node);
           current.next=new_n;
           current=new_n;
       }
       }

      void delete_node(int del_value){

          Node n=Head;
          Node np=null;


          while(n.data!=del_value){
          np=n;
          n=n.next;}
         
          np.next=n.next;
      }


      void Convert_to_doubly(){

      Node node=Head;

      while(node.next!=null){
          node=node.next;
          }

      node.next=Head;
      System.out.println("node.next"+node.next.data);


      }


       public void printList()
	{
		Node tnode = Head;
		while (tnode != null)
		{       System.out.print(tnode.data+" ");
			tnode = tnode.next;}}
